"""Utilities for the URL slug maker."""

from __future__ import annotations

import unicodedata


def _normalize_to_ascii(text: str) -> str:
    """Convert text to lowercase ASCII, removing accented characters."""
    lower_case_text = text.lower()
    accented_normalized = unicodedata.normalize("NFKD", lower_case_text)
    ascii_bytes = accented_normalized.encode("ascii", "ignore")
    return ascii_bytes.decode("ascii")


def _remove_apostrophes(text: str) -> str:
    """Remove all apostrophe characters from text."""
    return text.replace("'", "")


def _normalize_whitespace(text: str) -> str:
    """Collapse consecutive whitespace characters into single spaces."""
    result = []
    previous_was_whitespace = False

    for char in text:
        if char.isspace():
            if not previous_was_whitespace:
                result.append(" ")
                previous_was_whitespace = True
        else:
            result.append(char)
            previous_was_whitespace = False

    return "".join(result)


def _replace_punctuation_with_hyphens(text: str) -> str:
    """Replace non-alphanumeric characters with hyphens, no duplicates."""
    result = []
    previous_was_hyphen = False

    for char in text:
        if char.isascii() and char.isalnum():
            result.append(char)
            previous_was_hyphen = False
        elif not previous_was_hyphen:
            result.append("-")
            previous_was_hyphen = True

    return "".join(result)


def _split_into_words(text: str) -> list[str]:
    """Split text on hyphens into a list of words."""
    words = []
    current_word = []

    for char in text:
        if char == "-":
            if current_word:
                words.append("".join(current_word))
                current_word = []
        else:
            current_word.append(char)

    if current_word:
        words.append("".join(current_word))

    return words


def _truncate_to_max_length(words: list[str], max_length: int) -> str:
    """Join words with hyphens, truncating at word boundaries to max_length."""
    result_parts = []

    for word in words:
        if not result_parts:
            candidate = word
        else:
            candidate = "-".join(result_parts) + "-" + word

        if len(candidate) > max_length:
            break

        result_parts.append(word)

    return "-".join(result_parts)


def create_url_slug(title: str) -> str:
    """Create a URL slug from a title."""
    # Step 1: Normalize to ASCII (lowercase and remove accents)
    ascii_title = _normalize_to_ascii(title)

    # Step 2: Remove apostrophes
    no_apostrophes = _remove_apostrophes(ascii_title)

    # Step 3: Normalize whitespace
    normalized_whitespace = _normalize_whitespace(no_apostrophes)

    # Step 4: Replace punctuation with hyphens
    with_hyphens = _replace_punctuation_with_hyphens(normalized_whitespace)

    # Step 5: Strip leading/trailing hyphens
    trimmed = with_hyphens.strip("-")

    # Step 6: Split into words
    words = _split_into_words(trimmed)

    # Step 7: Truncate to maximum length at word boundaries
    return _truncate_to_max_length(words, 100)
